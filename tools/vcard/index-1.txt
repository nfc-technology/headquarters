        const gsTemplate = `function getOrCreateSheet() {
  var props = PropertiesService.getScriptProperties();
  var sheetId = props.getProperty('DB_SHEET_ID');
  var ss;
  
  if (sheetId) {
    try { ss = SpreadsheetApp.openById(sheetId); } 
    catch(e) { ss = null; }
  }
  
  if (!ss) {
    ss = SpreadsheetApp.create("vCard Profiles Database");
    props.setProperty('DB_SHEET_ID', ss.getId());
  }
  
  return ss.getActiveSheet();
}

function doPost(e) {
  try {
    var sheet = getOrCreateSheet();
    var data = JSON.parse(e.postData.contents);
    
    // Add Timestamp as the first column
    var payload = { "Timestamp": new Date().toISOString() };
    for (var key in data) { payload[key] = data[key]; }
    
    var headers = [];
    if (sheet.getLastRow() === 0) {
      // First time running: create headers from the payload keys
      headers = Object.keys(payload);
      sheet.appendRow(headers);
    } else {
      // Existing sheet: read headers, add new ones if necessary
      headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
      var newHeaders = [];
      Object.keys(payload).forEach(function(key) {
        if (headers.indexOf(key) === -1) {
          headers.push(key);
          newHeaders.push(key);
        }
      });
      if (newHeaders.length > 0) {
        sheet.getRange(1, 1, 1, headers.length).setValues([headers]);
      }
    }
    
    // Map data to the correct columns
    var row = headers.map(function(header) {
      var val = payload[header];
      if (val === undefined || val === null) return "";
      
      // Google Sheets cell character limit is 50,000. 
      // Base64 images can easily exceed this, which crashes the script.
      if (typeof val === 'string' && val.length > 49000) {
        return val.substring(0, 49000) + "... [Truncated to fit Sheets limits]";
      }
      return val;
    });
    
    sheet.appendRow(row);
    
    return ContentService.createTextOutput(JSON.stringify({"status": "success"})).setMimeType(ContentService.MimeType.JSON);
  } catch(error) {
    return ContentService.createTextOutput(JSON.stringify({"error": error.toString()})).setMimeType(ContentService.MimeType.JSON);
  }
}

function doOptions(e) {
  var headers = {
    "Access-Control-Allow-Origin": "*",
    "Access-Control-Allow-Methods": "POST, OPTIONS",
    "Access-Control-Allow-Headers": "Content-Type",
    "Access-Control-Max-Age": "86400"
  };
  return ContentService.createTextOutput("").setHeaders(headers);
}`;

        const formatImageForVCard = (base64Data, mimeType) => {
            const maxLineLength = 75;
            const prefix = `PHOTO;ENCODING=b;TYPE=${mimeType.toUpperCase()}:`;
            const firstLineSpace = maxLineLength - prefix.length;
            let result = prefix + base64Data.substring(0, firstLineSpace);
            let remainingData = base64Data.substring(firstLineSpace);
            
            while (remainingData.length > 0) {
                const chunk = remainingData.substring(0, maxLineLength - 1);
                result += '\r\n ' + chunk;
                remainingData = remainingData.substring(chunk.length);
            }
            return result;
        };

            if (logoData.base64 && logoData.type) {
                vcfContent.push(formatImageForVCard(logoData.base64, logoData.type).replace('PHOTO', 'LOGO'));
            }

            let waLink = formatWhatsAppUrl(getVal('social_whatsapp'));

            const socialProfiles = [ 
                {type: 'linkedin', value: getVal('social_linkedin')}, {type: 'twitter', value: getVal('social_twitter')},

        document.getElementById('reset-color-btn')?.addEventListener('click', () => {
            const picker = document.getElementById('primary-color-picker');
            if(picker) picker.value = '#DF1783';
            document.documentElement.style.setProperty('--primary-color', '#DF1783');
        });

        document.getElementById('save-settings-btn')?.addEventListener('click', () => {
            const picker = document.getElementById('primary-color-picker');
            const webhookUrl = document.getElementById('webhook-url');
            if(picker) localStorage.setItem('vcard-primary-color', picker.value);
            if(webhookUrl) localStorage.setItem('vcard-webhook', webhookUrl.value);
            toggleSettingsModal();
            showToast(currentLang === 'es' ? 'Configuración guardada' : 'Settings saved', 'success');
        });

        document.getElementById('copy-gs-code')?.addEventListener('click', () => {
            const codeBlock = document.getElementById('gs-code-block');
            if(codeBlock) {
                codeBlock.select();
                document.execCommand('copy');
                showToast('Code copied to clipboard!', 'success');
            }
        });

        // Load Settings on start
        const savedWebhook = localStorage.getItem('vcard-webhook');
        if(savedWebhook && document.getElementById('webhook-url')) {
            document.getElementById('webhook-url').value = savedWebhook;
        }

        // Form & Main Actions
        document.getElementById('vcard-form')?.addEventListener('input', updatePreview);
        document.getElementById('save-profile-btn')?.addEventListener('click', saveCurrentProfile);
        document.getElementById('download-btn')?.addEventListener('click', downloadVCF);
        
        document.getElementById('photo-upload')?.addEventListener('change', (e) => handleFileChange(e.target.files[0], photoData, 'preview-photo'));
        document.getElementById('logo-upload')?.addEventListener('change', (e) => handleFileChange(e.target.files[0], logoData, 'preview-logo'));
        
        document.getElementById('new-profile-btn')?.addEventListener('click', () => {
            clearForm();
            toggleProfilesPanel();
        });
        
        document.getElementById('profiles-list')?.addEventListener('click', (e) => {
            const target = e.target.closest('li');
            const deleteButton = e.target.closest('button');
            if (deleteButton && deleteButton.dataset.deleteId) {
                deleteProfile(deleteButton.dataset.deleteId);
            } else if (target && target.dataset.id) {
                loadProfile(target.dataset.id, true);
            }
        });

        // Keyboard Shortcuts
        document.addEventListener('keydown', (e) => {
            if (e.ctrlKey || e.metaKey) {
                if (e.key.toLowerCase() === 's') {
                    e.preventDefault();
                    saveCurrentProfile();
                } else if (e.key.toLowerCase() === 'd') {
                    e.preventDefault();
                    document.getElementById('download-btn')?.click();
                } else if (e.key === ',') {
                    e.preventDefault();
                    toggleSettingsModal();
                }
            }
        });

        // INIT
        applyLanguage(currentLang);
        loadProfilesFromStorage();
        if(profiles.length > 0 && profiles[0]?.id) loadProfile(profiles[0].id);
        else clearForm();

    });
